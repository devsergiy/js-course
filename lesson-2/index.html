Типы (types)

Простые типы

  • string
  • number
  • boolean
  • null
  • undefined
Когда вы взаимодействуете с простым типом, вы взаимодействуете непосредственно с его значением в памяти. 

var foo = 1,
    bar = foo;

bar = 9;

console.log(foo, bar); // => 1, 9. foo не изменился

Сложные типы

  • object
  • array
  • function
Когда вы взаимодействуете со сложным типом, вы взаимодействуете с ссылкой на его значение в памяти. 

var foo = [1, 2],
    bar = foo;

bar[0] = 9; // bar теперь ссылается на foo

console.log(foo[0], bar[0]); // => 9, 9.

Условные операторы

Оператор IF


if (year != 2011) {
	console.log('Не тот год');
} else if (year == 2012) {
	console.log('Не этот год');
} else {
	console.log('Всякий');
}

Тернарная форма оператора IF


var x = (year != 2011)  ? 'Не тот год':  'Всякий';
console.log(x);

Оператор SWITCH


switch (year) {
	case 2011:
		console.log('2011');
	case 2012:
		console.log('2012');
		break;
	default:
		console.log('Всякий');
}

Операторы сравнения >, <, <=, =>


6 > 4;		// true
9 < 5;		// false
8 >= -2;	// true
3 == 4;		// false
12 != 4;	// true
10 <= 10;	// true

						

4 == "4";		// true

Сложные условия


true && false;		// false
true && true;		// true
false && false;		// false
false || true;		// true
false || false;		// false
true || true;			// true

&& - бинарное "И", возвращает true, если оба значение true

|| - бинарное "ИЛИ", возвращает true, если хотя бы одно значение true

Число 0, пустая строка "", null и undefined, а также NaN являются false,


var z, n;

n = 'number' * 2;

var x = 0 || '' || null || z || n || 2015;

console.log(x);
						

//> 2015
